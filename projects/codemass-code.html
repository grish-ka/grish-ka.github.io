 24from utime import sleep

 26MORSE_CODE_DICT = { 'A':'.-', 'B':'-...',
 27    'C':'-.-.', 'D':'-..', 'E':'.',
 28    'F':'..-.', 'G':'--.', 'H':'....',
 29    'I':'..', 'J':'.---', 'K':'-.-',
 30    'L':'.-..', 'M':'--', 'N':'-.',
 31    'O':'---', 'P':'.--.', 'Q':'--.-',
 32    'R':'.-.', 'S':'...', 'T':'-',
 33    'U':'..-', 'V':'...-', 'W':'.--',
 34    'X':'-..-', 'Y':'-.--', 'Z':'--..',
 35    '1':'.----', '2':'..---', '3':'...--',
 36    '4':'....-', '5':'.....', '6':'-....',
 37    '7':'--...', '8':'---..', '9':'----.',
 38    '0':'-----', ', ':'--..--', '.':'.-.-.-',
 39    '?':'..--..', '/':'-..-.', '-':'-....-',
 40    '(':'-.--.', ')':'-.--.-' }

 43led = Pin("LED", Pin.OUT)
 44red = Pin(18, Pin.OUT)
 45yellow = Pin(19, Pin.OUT)
 46green = Pin(20, Pin.OUT)
 47
 48def encrypt(message):
 49    cipher = ''
 50    for letter in message:
 51        if letter != ' ':
 52            cipher += MORSE_CODE_DICT[letter] + ' '
 53        else:
 54            cipher += ' '
 55    return cipher
 56
 57def decrypt(message):
 58    message += ' '
 59    decipher = ''
 60    citext = ''
 61    for letter in message:
 62        if (letter != ' '):
 63            i = 0
 64            citext += letter
 65        else:
 66            i += 1
 67            if i == 2 :
 68                decipher += ' '
 69            else:
 70                if citext:
 71                    decipher += list(MORSE_CODE_DICT.keys())[list(MORSE_CODE_DICT
 72                    .values()).index(citext)]
 73                citext = ''
 74    return decipher.strip()

 76def startup():
 77    led.on()
 78    sleep(1)
 79    red.on()
 80    sleep(1)
 81    yellow.on()
 82    sleep(1)
 83    green.on()
 84    sleep(1)
 85    led.off()
 86    red.off()
 87    yellow.off()
 88    green.off()
 89
 90if __name__ == "__main__":
 91    try:
 92        startup()
 93        message = "HELLO WORLD"
 94        result = encrypt(message)
 95        print (result)
 96        for symbol in result:
 97            if symbol == '.':
 98                led.on()
 99                red.on()
100                sleep(0.2)
101                led.off()
102                red.off()
103            elif symbol == '-':
104                led.on()
105                yellow.on()
106                sleep(0.6)
107                led.off()
108                yellow.off()
109            else:
110                sleep(0.6)  # Space between letters
111            sleep(0.2)  # Space between symbols
112        green.on()
113        print("Message \"%s\" transmitted in Morse code." % decrypt(result))
114        sleep(5)
115        green.off()
116
117    except KeyboardInterrupt:
118        print ("Transmission interrupted.")
119        led.off()
120        red.off()
121        yellow.off()
122        green.off()
123        red.on()
124        sleep(3)
125        yellow.on()
126        green.on()
127        led.on()
128        sleep(5)
129        red.off()
130        yellow.off()
131        green.off()
